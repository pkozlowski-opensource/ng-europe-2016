<input [(ngModel)]="state" [ngbTypeahead]="findState" />
 findState = (userInput$: Observable<string>) => userInput$
  .debounceTime(200)
  .distinctUntilChanged()
  .map(term => term.length < 2 ? [] : states.filter(state => {
    	return new RegExp(term, 'gi').test(state)).splice(0, 10);
    });
